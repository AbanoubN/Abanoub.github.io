.text ; program start

.global _main ; define entry point

_main:

NOP


bis.b #11111111b, &P3DIR

mov.w #11111111b, &P3OUT

bis.b #0000001b,&P1DIR

mov.w #0, R10

mov.b #8, R11

mov.b #0, R12



Mainloop:

add.b #1b,R10

mov.b R10,R12

and.b R11,R12

cmp.b R12,R11

jeq Mainloop

xor.b #0xff,R10

mov.b R10,&P3OUT

xor.b #0xff,R10


jmp Mainloop

NOP
